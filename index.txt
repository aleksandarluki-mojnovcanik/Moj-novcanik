        ucitajPodatke();
    }

    // Tiha migracija: Ako postoje podaci iz stare verzije, dodeli ih prvom kreiranom profilu da se ne izgube
    function proveriIPrebaciStarePodatke(profil) {
        let stariPodaci = localStorage.getItem("mojNovcanik_podaci");
        let noviPodaci = localStorage.getItem("mojNovcanik_podaci_" + profil);
        
        if (!noviPodaci && stariPodaci) {
            localStorage.setItem("mojNovcanik_podaci_" + profil, stariPodaci);
            if(localStorage.getItem("mojNovcanik_kategorije")) localStorage.setItem("mojNovcanik_kategorije_" + profil, localStorage.getItem("mojNovcanik_kategorije"));
            if(localStorage.getItem("mojNovcanik_pocetnaStednja")) localStorage.setItem("mojNovcanik_pocetnaStednja_" + profil, localStorage.getItem("mojNovcanik_pocetnaStednja"));
            if(localStorage.getItem("mojNovcanik_pocetniDzep")) localStorage.setItem("mojNovcanik_pocetniDzep_" + profil, localStorage.getItem("mojNovcanik_pocetniDzep"));
            if(localStorage.getItem("mojNovcanik_pocetniDug")) localStorage.setItem("mojNovcanik_pocetniDug_" + profil, localStorage.getItem("mojNovcanik_pocetniDug"));
            
            localStorage.removeItem("mojNovcanik_podaci");
        }
    }

    function getKey(base) {
        return base + "_" + trenutniProfil;
    }

    function getLokalniDatum() {
        const d = new Date();
        return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;
    }

    function ucitajPodatke() {
        let sacuvano = localStorage.getItem(getKey("mojNovcanik_podaci"));
        if (sacuvano !== null) bazaPodataka = JSON.parse(sacuvano).map(st => { if (!st.datum) st.datum = getLokalniDatum(); if (!st.tip) st.tip = 'trosak'; return st; });
        else bazaPodataka = [];

        let sacuvaneKat = localStorage.getItem(getKey("mojNovcanik_kategorije"));
        listaKategorija = sacuvaneKat !== null ? JSON.parse(sacuvaneKat) : [...defaultKategorije];

        pocetnaStednja = localStorage.getItem(getKey("mojNovcanik_pocetnaStednja")) !== null ? Number(localStorage.getItem(getKey("mojNovcanik_pocetnaStednja"))) : 0;
        pocetniDzep = localStorage.getItem(getKey("mojNovcanik_pocetniDzep")) !== null ? Number(localStorage.getItem(getKey("mojNovcanik_pocetniDzep"))) : 0;
        pocetniDug = localStorage.getItem(getKey("mojNovcanik_pocetniDug")) !== null ? Number(localStorage.getItem(getKey("mojNovcanik_pocetniDug"))) : 0;

        let poznateKat = new Set(listaKategorija.map(k => k.naziv));
        bazaPodataka.forEach(st => {
            if ((st.tip === 'trosak' || st.tip === 'dug_otplata') && st.kat && !poznateKat.has(st.kat)) {
                listaKategorija.push({ naziv: st.kat, ikona: '📌' }); poznateKat.add(st.kat);
            }
        });

        osveziFilterMeseca(); iscrtajSve();
    }

    function promeniPocetniStek() {
        let unos = prompt("Unesi početni iznos koji već imaš u sefu/banki:", pocetnaStednja);
        if (unos !== null && !isNaN(Number(unos))) { pocetnaStednja = Number(unos); localStorage.setItem(getKey("mojNovcanik_pocetnaStednja"), pocetnaStednja); iscrtajSve(); }
    }

    function promeniPocetniDzep() {
        let unos = prompt("Unesi keš koji si imao onog dana kada si počeo da koristiš aplikaciju:", pocetniDzep);
        if (unos !== null && !isNaN(Number(unos))) { pocetniDzep = Number(unos); localStorage.setItem(getKey("mojNovcanik_pocetniDzep"), pocetniDzep); iscrtajSve(); }
    }

    function promeniPocetniDug() {
        let unos = prompt("Unesi ukupni dug koji VEC DUGOVAO pre aplikacije:", pocetniDug);
        if (unos !== null && !isNaN(Number(unos))) { pocetniDug = Number(unos); localStorage.setItem(getKey("mojNovcanik_pocetniDug"), pocetniDug); iscrtajSve(); }
    }

    function toggleFioka(tip) { fiokeStanje[tip] = !fiokeStanje[tip]; iscrtajSve(); }
    function prebaciKategoriju(nazivKat) { selektovanaKategorija = (selektovanaKategorija === nazivKat) ? "" : nazivKat; iscrtajSve(); }

    function obrisiKategoriju(imeKat, event) {
        event.stopPropagation();
        if (!confirm(`Brišeš kategoriju "${imeKat}"?`)) return;
        let vezani = bazaPodataka.filter(st => (st.tip === 'trosak' || st.tip === 'dug_otplata') && st.kat === imeKat);
        if (vezani.length > 0) {
            alert(`Troškovi prebačeni u "📦 Ostalo".`);
            if (!listaKategorija.some(k => k.naziv === 'Ostalo')) listaKategorija.push({ naziv: 'Ostalo', ikona: '📦' });
            bazaPodataka = bazaPodataka.map(st => { if ((st.tip === 'trosak' || st.tip === 'dug_otplata') && st.kat === imeKat) st.kat = 'Ostalo'; return st; });
            localStorage.setItem(getKey("mojNovcanik_podaci"), JSON.stringify(bazaPodataka));
        }
        listaKategorija = listaKategorija.filter(k => k.naziv !== imeKat);
        localStorage.setItem(getKey("mojNovcanik_kategorije"), JSON.stringify(listaKategorija));
        if (selektovanaKategorija === imeKat) selektovanaKategorija = "";
        iscrtajSve();
    }

    function urediTransakciju(idZaUredjivanje) {
        let st = bazaPodataka.find(item => Number(item.id) === Number(idZaUredjivanje));
        if (!st || st.type === 'uploaded file') return; 

        if (st.tip === 'prihod') {
            document.getElementById("prihod-opis").value = st.naziv; document.getElementById("prihod-iznos").value = st.cena; fiokeStanje.prihod = true; 
        } else if (st.tip === 'stednja' || st.tip === 'stednja_kamata') {
            document.getElementById("stednja-opis").value = st.naziv; document.getElementById("stednja-iznos").value = st.cena; fiokeStanje.stednja = true;
        } else if (st.tip === 'dug_plus' || st.tip === 'dug_otplata') {
            document.getElementById("dug-opis").value = st.naziv; document.getElementById("dug-iznos").value = st.cena; 
            if(st.tip === 'dug_otplata') document.getElementById("dug-kat-select").value = st.kat;
            fiokeStanje.dug = true;
        } else {
            document.getElementById("trosak-opis").value = st.naziv; document.getElementById("trosak-iznos").value = st.cena; selektovanaKategorija = st.kat;
        }

        bazaPodataka = bazaPodataka.filter(item => Number(item.id) !== Number(idZaUredjivanje));
        localStorage.setItem(getKey("mojNovcanik_podaci"), JSON.stringify(bazaPodataka));
        osveziFilterMeseca(); iscrtajSve();
    }

    function kreirajKategoriju() {
        let ime = prompt("Naziv nove kategorije:");
        if (!ime || ime.trim() === "") return;
        let cisto = ime.trim();
        if (!listaKategorija.some(k => k.naziv.toLowerCase() === cisto.toLowerCase())) {
            listaKategorija.push({ naziv: cisto, ikona: '🏷️' }); localStorage.setItem(getKey("mojNovcanik_kategorije"), JSON.stringify(listaKategorija));
        }
        selektovanaKategorija = cisto; iscrtajSve(); 
    }

    function upisiDrobneSporeni() {
        bazaPodataka.push({ id: Date.now(), tip: 'stednja', datum: getLokalniDatum(), naziv: 'Drobné spoření (Auto)', cena: 100, kat: 'Štednja' });
        localStorage.setItem(getKey("mojNovcanik_podaci"), JSON.stringify(bazaPodataka));
        osveziFilterMeseca(); iscrtajSve();
    }

    function upisiTransakciju(tip) {
        let prefiks = tip;
        if (tip === 'dug_plus' || tip === 'dug_otplata') prefiks = 'dug';
        if (tip === 'stednja_kamata') prefiks = 'stednja';

        let poljeOpis = document.getElementById(`${prefiks}-opis`);
        let poljeIznos = document.getElementById(`${prefiks}-iznos`);
        let odredjeniDatum = getLokalniDatum();
        
        if (poljeOpis.value === "" || Number(poljeIznos.value) <= 0) {
            alert("Upiši tačne podatke!"); return;
        }

        if (tip === 'trosak' && selektovanaKategorija === "") {
            alert("Klikni prvo na neku ikonicu kategorije troška u kasi levo!"); return;
        }

        let dodeljenaKat = 'Razno';
        if(tip === 'trosak') dodeljenaKat = selektovanaKategorija;
        if(tip === 'stednja' || tip === 'stednja_kamata') dodeljenaKat = 'Štednja';
        if(tip === 'dug_plus') dodeljenaKat = 'Zajam';
        if(tip === 'dug_otplata') dodeljenaKat = document.getElementById("dug-kat-select").value;

        bazaPodataka.push({
            id: Date.now(), tip: tip, datum: odredjeniDatum,
            naziv: poljeOpis.value, cena: Number(poljeIznos.value), kat: dodeljenaKat
        });

        localStorage.setItem(getKey("mojNovcanik_podaci"), JSON.stringify(bazaPodataka));
        osveziFilterMeseca(); iscrtajSve();
        poljeOpis.value = ""; poljeIznos.value = ""; poljeOpis.focus();
    }

    function obrisiTransakciju(idZaBrisanje) {
        bazaPodataka = bazaPodataka.filter(st => Number(st.id) !== Number(idZaBrisanje));
        localStorage.setItem(getKey("mojNovcanik_podaci"), JSON.stringify(bazaPodataka));
        osveziFilterMeseca(); iscrtajSve();
    }

    function osveziFilterMeseca() {
        let selektor = document.getElementById("filter-mesec");
        let trenutnoIzabran = selektor.value;
        selektor.innerHTML = `<option value="svi">-- Svi meseci zajedno --</option>`;

        let setMeseca = new Set([getLokalniDatum().slice(0, 7)]);
        bazaPodataka.forEach(st => setMeseca.add(st.datum.slice(0, 7)));
        Array.from(setMeseca).sort().reverse().forEach(gMesec => {
            let [g, m] = gMesec.split("-");
            let opcija = document.createElement("option");
            opcija.value = gMesec; opcija.innerText = `${imenaMeseca[Number(m)-1]} ${g}`;
            selektor.appendChild(opcija);
        });
        selektor.value = (trenutnoIzabran && Array.from(selektor.options).some(o => o.value === trenutnoIzabran)) ? trenutnoIzabran : getLokalniDatum().slice(0, 7);
    }

    function formatirajDatumPrikaz(isoDatum) { let [g, m, d] = isoDatum.split("-"); return `${d}.${m}.${g}.`; }

    function iscrtajSve() {
        let izabranPeriod = document.getElementById("filter-mesec").value;
        let zP_mesec = 0; let zT_mesec = 0;
        let zP_kum = 0; let zT_kum = 0; let zS_kum = 0;
        let zS_svi_ikada = 0; 
        let zD_plus_svi = 0; let zD_minus_svi = 0; 

        let sakupljacPoKat = {};
        listaKategorija.forEach(k => { sakupljacPoKat[k.naziv] = 0; });

        let pFiokaStavke = []; let sFiokaStavke = []; let dFiokaStavke = []; let tDnoStavke = [];

        bazaPodataka.forEach(st => {
            let mesecTransakcije = st.datum.slice(0, 7);

            if (st.tip === 'stednja' || st.tip === 'stednja_kamata') zS_svi_ikada += st.cena;
            if (st.tip === 'dug_plus') zD_plus_svi += st.cena;
            if (st.tip === 'dug_otplata') zD_minus_svi += st.cena;

            if (izabranPeriod === "svi" || mesecTransakcije === izabranPeriod) {
                if (st.tip === 'prihod') { zP_mesec += st.cena; pFiokaStavke.push(st); }
                else if (st.tip === 'stednja' || st.tip === 'stednja_kamata') { sFiokaStavke.push(st); }
                else if (st.tip === 'dug_plus' || st.tip === 'dug_otplata') { dFiokaStavke.push(st); } 
                
                if (st.tip === 'trosak' || st.tip === 'dug_otplata') {
                    zT_mesec += st.cena;
                    if (sakupljacPoKat[st.kat] !== undefined) sakupljacPoKat[st.kat] += st.cena;
                    if (selektovanaKategorija !== "" && st.kat === selektovanaKategorija) tDnoStavke.push(st);
                }
            }

            if (izabranPeriod === "svi" || mesecTransakcije <= izabranPeriod) {
                if (st.tip === 'prihod') zP_kum += st.cena;
                if (st.tip === 'trosak' || st.tip === 'dug_otplata') zT_kum += st.cena; 
                // Samo obicna stednja izbija kes iz dzepa!
                if (st.tip === 'stednja') zS_kum += st.cena;
            }
        });

        let stanjeSefa = pocetnaStednja + zS_svi_ikada;
        let stanjeDuga = pocetniDug + zD_plus_svi - zD_minus_svi;
        let raspolozivoDzep = pocetniDzep + zP_kum - zT_kum - zS_kum; 

        document.getElementById("prikaz-prihodi").innerText = zP_mesec + " Kč";
        document.getElementById("prikaz-troskovi").innerText = zT_mesec + " Kč";
        document.getElementById("prikaz-stednja").innerText = stanjeSefa + " Kč";
        document.getElementById("prikaz-dug").innerText = stanjeDuga + " Kč";
        document.getElementById("prikaz-stanje").innerText = raspolozivoDzep + " Kč";

        document.querySelector(".ekran-stanje").style.background = raspolozivoDzep < 0 ? "linear-gradient(145deg, #b71c1c, #f44336)" : "linear-gradient(145deg, #0d47a1, #1976d2)";
        document.querySelector(".ekran-dug").style.background = stanjeDuga > 0 ? "linear-gradient(145deg, #4a148c, #8e24aa)" : "linear-gradient(145deg, #2e2e2e, #444)";

        let dugSel = document.getElementById("dug-kat-select");
        if(dugSel) {
            let staraKat = dugSel.value;
            dugSel.innerHTML = listaKategorija.map(k => `<option value="${k.naziv}">${k.ikona || '🏷️'} ${k.naziv}</option>`).join('');
            if(staraKat && listaKategorija.some(k => k.naziv === staraKat)) dugSel.value = staraKat;
        }

        let fP = document.getElementById("fioka-prihod"); let btnP = document.getElementById("btn-toggle-p");
        if(fiokeStanje.prihod) {
            fP.style.display = "block"; btnP.innerHTML = "❌ SAKRIJ"; btnP.style.color = "#ff5252"; btnP.style.borderColor = "#ff5252";
            document.getElementById("lista-prihod-unutra").innerHTML = pFiokaStavke.length === 0 ? `<div class="info-prazno">Prazno</div>` : pFiokaStavke.map(st => `<li class="li-prihod"><div class="levo-st"><span class="datum-tag">${formatirajDatumPrikaz(st.datum)}</span><span class="naslov-stavke">${st.naziv}</span></div><div class="desno-st"><span class="cena-zelena">+${st.cena}</span><button class="uredi-btn" onclick="urediTransakciju(${st.id})">✏️</button><button class="obrisi-btn" onclick="obrisiTransakciju(${st.id})">X</button></div></li>`).join('');
        } else { fP.style.display = "none"; btnP.innerHTML = "✏️ LISTA PRIHODA"; btnP.style.color = "#4caf50"; btnP.style.borderColor = "#4caf50"; }

        let fS = document.getElementById("fioka-stednja"); let btnS = document.getElementById("btn-toggle-s");
        if(fiokeStanje.stednja) {
            fS.style.display = "block"; btnS.innerHTML = "❌ SAKRIJ"; btnS.style.color = "#ff5252"; btnS.style.borderColor = "#ff5252";
            document.getElementById("lista-stednja-unutra").innerHTML = sFiokaStavke.length === 0 ? `<div class="info-prazno">Prazno</div>` : sFiokaStavke.map(st => `<li class="li-stednja"><div class="levo-st"><span class="datum-tag">${formatirajDatumPrikaz(st.datum)}</span><span class="naslov-stavke">${st.naziv}</span></div><div class="desno-st"><span class="${st.tip === 'stednja_kamata' ? 'cena-zelena' : 'cena-zlatna'}">${st.tip === 'stednja_kamata' ? '+' : ''}${st.cena}</span><button class="uredi-btn" onclick="urediTransakciju(${st.id})">✏️</button><button class="obrisi-btn" onclick="obrisiTransakciju(${st.id})">X</button></div></li>`).join('');
        } else { fS.style.display = "none"; btnS.innerHTML = "✏️ LISTA ŠTEDNJE"; btnS.style.color = "#ff9800"; btnS.style.borderColor = "#ff9800"; }

        let fD = document.getElementById("fioka-dug"); let btnD = document.getElementById("btn-toggle-d");
        if(fiokeStanje.dug) {
            fD.style.display = "block"; btnD.innerHTML = "❌ SAKRIJ"; btnD.style.color = "#ff5252"; btnD.style.borderColor = "#ff5252";
            document.getElementById("lista-dug-unutra").innerHTML = dFiokaStavke.length === 0 ? `<div class="info-prazno">Prazno</div>` : dFiokaStavke.map(st => `<li class="${st.tip === 'dug_plus' ? 'li-dug-plus' : 'li-dug-minus'}"><div class="levo-st"><span class="datum-tag">${formatirajDatumPrikaz(st.datum)}</span><span class="naslov-stavke">${st.naziv}</span></div><div class="desno-st"><span class="${st.tip === 'dug_plus' ? 'cena-ljubicasta' : 'cena-zelena'}">${st.tip === 'dug_plus' ? '+' : '-'}${st.cena}</span><button class="uredi-btn" onclick="urediTransakciju(${st.id})">✏️</button><button class="obrisi-btn" onclick="obrisiTransakciju(${st.id})">X</button></div></li>`).join('');
        } else { fD.style.display = "none"; btnD.innerHTML = "✏️ LISTA DUGOVA"; btnD.style.color = "#ab47bc"; btnD.style.borderColor = "#ab47bc"; }

        document.getElementById("deck-kategorija-unos").innerHTML = listaKategorija.map(k => `<div class="chip ${k.naziv === selektovanaKategorija ? 'aktivan' : ''}" onclick="prebaciKategoriju('${k.naziv}')"><span class="del-kat" onclick="obrisiKategoriju('${k.naziv}', event)">×</span><span>${k.ikona || '🏷️'}</span> ${k.naziv}</div>`).join('') + `<div class="chip chip-plus" onclick="kreirajKategoriju()">+ Nova</div>`;

        document.getElementById("stat-mreza").innerHTML = listaKategorija.map(k => `<div class="stat-kartica ${selektovanaKategorija === k.naziv ? 'selektovana' : ''}" onclick="prebaciKategoriju('${k.naziv}')"><span class="del-kat" onclick="obrisiKategoriju('${k.naziv}', event)">×</span><span class="ikona">${k.ikona || '🏷️'}</span><span class="ime">${k.naziv}</span><span class="iznos ${sakupljacPoKat[k.naziv] === 0 ? 'nula' : ''}">${sakupljacPoKat[k.naziv] || 0} Kč</span></div>`).join('') + `<div class="stat-kartica kartica-plus" onclick="kreirajKategoriju()"><span class="ikona">➕</span><span class="ime">Nova</span></div>`;

        let detBox = document.getElementById("detalji-box");
        if (selektovanaKategorija === "") detBox.style.display = "none";
        else {
            detBox.style.display = "block";
            document.getElementById("naslov-detalja").innerText = `Spisak troškova za: ${selektovanaKategorija}`;
            document.getElementById("lista-detalja").innerHTML = tDnoStavke.length === 0 ? `<div class="info-prazno">Nema unosa u ovoj kategoriji.</div>` : tDnoStavke.map(st => `<li class="li-trosak"><span class="naslov-stavke">${st.naziv}</span><div class="desno-st"><span class="cena-crvena">-${st.cena} Kč</span><button class="uredi-btn" onclick="urediTransakciju(${st.id})">✏️</button><button class="obrisi-btn" onclick="obrisiTransakciju(${st.id})">X</button></div></li>`).join('');
        }
    }

    function exportData() {
        let data = {
            baza: localStorage.getItem(getKey("mojNovcanik_podaci")),
            kategorije: localStorage.getItem(getKey("mojNovcanik_kategorije")),
            stednja: localStorage.getItem(getKey("mojNovcanik_pocetnaStednja")),
            dzep: localStorage.getItem(getKey("mojNovcanik_pocetniDzep")),
            dug: localStorage.getItem(getKey("mojNovcanik_pocetniDug"))
        };
        let blob = new Blob([JSON.stringify(data)], {type: "application/json"});
        let a = document.createElement("a");
        a.href = URL.createObjectURL(blob);
        a.download = "novcanik_" + trenutniProfil + "_backup.json";
        a.click();
    }
    
    function importData(event) {
        let file = event.target.files[0];
        let reader = new FileReader();
        reader.onload = function(e) {
            let data = JSON.parse(e.target.result);
            if(data.baza) localStorage.setItem(getKey("mojNovcanik_podaci"), data.baza);
            if(data.kategorije) localStorage.setItem(getKey("mojNovcanik_kategorije"), data.kategorije);
            if(data.stednja) localStorage.setItem(getKey("mojNovcanik_pocetnaStednja"), data.stednja);
            if(data.dzep) localStorage.setItem(getKey("mojNovcanik_pocetniDzep"), data.dzep);
            if(data.dug) localStorage.setItem(getKey("mojNovcanik_pocetniDug"), data.dug);
            location.reload();
        };
        reader.readAsText(file);
    }
